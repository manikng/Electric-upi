// ── POST /api/bookings/[id]/accept ──
// Charger host accepts a pending booking request.
// Generates secure 6-digit OTP code using crypto.randomInt.
export async function POST(
  request: Request,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const { id } = await params;

    // 1. Verify user session
    const supabase = await createSupabaseServerClient();
    const {
      data: { user },
    } = await supabase.auth.getUser();

    if (!user) {
      return NextResponse.json({ error: "Unauthorized. Please sign in." }, { status: 401 });
    }

    // 2. Fetch booking and parent charger details
    const [booking] = await db
      .select({
        bookingId: bookings.id,
        status: bookings.status,
        chargerId: bookings.chargerId,
        hostId: chargers.hostId,
      })
      .from(bookings)
      .innerJoin(chargers, eq(bookings.chargerId, chargers.id))
      .where(eq(bookings.id, id))
      .limit(1);

    if (!booking) {
      return NextResponse.json({ error: "Booking not found." }, { status: 404 });
    }

    // 3. Verify that caller is the host of this charger (adversarial guard)
    if (booking.hostId !== user.id) {
      return NextResponse.json({ error: "Forbidden. You are not the host for this charger." }, { status: 403 });
    }

    // 4. Verify status is pending
    if (booking.status !== "pending_host_accept") {
      return NextResponse.json(
        { error: `Cannot accept booking in its current state: '${booking.status}'` },
        { status: 400 }
      );
    }

    // 5. Generate secure 6-digit mutual nonce and expiration (15 minutes from now)
    const nonce = crypto.randomInt(100000, 1000000).toString().padStart(6, "0");
    const expiry = new Date(Date.now() + 15 * 60 * 1000); // 15 mins

    // 6. Conditional update: set nonce and handshake fields only if booking is still pending
    const result = await db
      .update(bookings)
      .set({
        status: "awaiting_handshake",
        nonce: nonce,
        nonceExpiresAt: expiry,
        nonceUsed: false,
        hostConfirmed: false,
        driverConfirmed: false,
        nonceGeneratedAt: new Date(),
        nonceGeneratedBy: user.id,
      })
      .where(and(eq(bookings.id, id), eq(bookings.status, "pending_host_accept")));

    // verify update affected a row
    // drizzle returns a result we can inspect via .rowCount in some drivers; fall back to selecting
    const [updatedBooking] = await db.select().from(bookings).where(eq(bookings.id, id)).limit(1);
    if (!updatedBooking || updatedBooking.status !== "awaiting_handshake") {
      return NextResponse.json({ error: "Failed to accept booking (state changed)." }, { status: 409 });
    }

    return NextResponse.json({
      success: true,
      status: "awaiting_driver_arrival",
      message: "Booking request accepted. Awaiting driver arrival code verification.",
    }, { status: 200 });
  } catch (error) {
    console.error("POST /api/bookings/[id]/accept error:", error);
    return NextResponse.json({ error: "Internal server error" }, { status: 500 });
  }
}

export async function POST(
  request: Request,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const { id } = await params;

    // 1. Verify user session
    const supabase = await createSupabaseServerClient();
    const {
      data: { user },

      .select({
        bookingId: bookings.id,
        status: bookings.status,
        driverId: bookings.driverId,
        pricePerKwh: chargers.pricePerKwh,
        hostId: chargers.hostId,
      })
      .from(bookings)
      .innerJoin(chargers, eq(bookings.chargerId, chargers.id))
      .where(eq(bookings.id, id))
      .limit(1);

    if (!booking) {
      return NextResponse.json({ error: "Booking not found." }, { status: 404 });
    }

    // 3. Authorization Guard: caller must be driver or host
    const isDriver = booking.driverId === user.id;
    const isHost = booking.hostId === user.id;

    if (!isDriver && !isHost) {
      return NextResponse.json({ error: "Forbidden. You are not a party to this booking." }, { status: 403 });
    }

    // 4. Verify booking is verified
    if (booking.status !== "charging") {
      return NextResponse.json(
        { error: `Cannot end charging session. Booking status is '${booking.status}' (expected 'charging')` },
        { status: 400 }
      );
    }

    // 5. Calculate energy consumption metrics
    const energyKwhSafe = "10.500"; // Mock 10.5 kWh consumption
    const price = parseFloat(booking.pricePerKwh);
    const cost = price * parseFloat(energyKwhSafe);

    // 6. Update booking state: set endedAt to current time, status to 'completed', record energy
    await db
      .update(bookings)
      .set({
        status: "completed",
        endedAt: new Date(),
        energyKwh: energyKwhSafe,
      })
      .where(eq(bookings.id, id));

    return NextResponse.json({
      success: true,
      status: "completed",
      energyKwh: 10.5,
      cost: parseFloat(cost.toFixed(2)),
      message: "Charging session completed successfully. Pending UPI payment split.",
    }, { status: 200 });

    console.error("POST /api/bookings/[id]/end error:", error);
    return NextResponse.json({ error: "Internal server error" }, { status: 500 });
